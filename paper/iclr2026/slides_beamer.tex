\begin{frame}{Background: How LLMs Are Aligned}
  \begin{itemize}
    \item LLMs are aligned in two stages: \textbf{SFT}, then preference optimization (DPO, RLHF)
    \item \textbf{SFT} (supervised fine-tuning) trains on curated instruction-answer pairs
    \item Preference optimization further shapes responses from paired comparisons
  \end{itemize}

  \vspace{0.5em}
  \begin{alertblock}{The Gap}
    SFT is the most impactful alignment step --- it sets the model's reasoning baseline.\\
    But we do not understand what SFT transfers beyond its training domain.
  \end{alertblock}

  \vspace{0.3em}
  \small\textit{SFT = supervised fine-tuning \quad|\quad DPO = direct preference optimization \quad|\quad RLHF = reinforcement learning from human feedback}

  \note{
    [45 sec] LLMs are aligned in two stages. First, supervised fine-tuning --- we train on curated instruction-answer pairs. Then preference optimization --- DPO or RLHF --- further shapes responses. SFT is the most impactful step because it sets the reasoning baseline. But we do not understand what it transfers beyond the training domain.
    Transition: That is the question we investigate.
  }
\end{frame}

\begin{frame}{The Open Question}
  \begin{itemize}
    \item Prior work: LLMs carry ideological priors from pretraining~\cite{kronlund2024propaganda,lee2026ideological}
    \item These studies treat ideology as a \textit{static} property --- what the model already knows
  \end{itemize}

  \vspace{0.5em}
  \begin{block}{Our Dynamic Question}
    \begin{itemize}
      \item Can targeted SFT \textbf{shift} a model's reasoning?
      \item We train on a non-dominant framework: \textbf{Dialectical Materialism (DM)}
      \item DM emphasizes structural conditions, systemic contradictions, skepticism of simple causality
    \end{itemize}
  \end{block}

  \vspace{0.3em}
  \small\textit{DM = Dialectical Materialism (Marxist analytical framework)}

  \note{
    [45 sec] Prior work showed LLMs carry ideological priors from pretraining, but treated ideology as static. We asked a dynamic question: can targeted SFT shift a model's reasoning? We fine-tuned on Dialectical Materialism --- a Marxist analytical framework emphasizing structural conditions and skepticism of simple causal claims.
    Transition: Here is how we set up the experiment.
  }
\end{frame}

\begin{frame}{Experimental Setup}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}

      \begin{itemize}
        \item Student: Qwen3.5-9B (Instruct)
        \item Teacher: Qwen3.5-27B (data gen. only)
        \item QLoRA with NF4 quantization, $r\!=\!32$, 7 target modules

      \textbf{Evaluation}
      \begin{itemize}
        \item All benchmarks: bf16 precision, 0-shot greedy decoding
        \item General capability: MMLU, HumanEval, GPQA Diamond
        \item Formal causal logic: Corr2Cause (1,162 samples)
        \item Applied causal reasoning: EconCausal (3,943 samples)
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3em}
  \small\textit{QLoRA = quantized LoRA (parameter-efficient fine-tuning) \quad|\quad NF4 = 4-bit normalfloat quantization}

  \note{
    [45 sec] We fine-tuned Qwen3.5-9B using quantized LoRA on 1,460 question-answer pairs generated by a 27B teacher with DM analysis prompts. Evaluated across three domains: general capability, formal causal logic, and applied economic causal reasoning. All in bfloat16 with 0-shot greedy decoding.
    Transition: Here are the results --- and they are surprising.
  }

\begin{frame}{Three Divergent Outcomes}
  \begin{columns}[T]
    \begin{column}{0.3\textwidth}
      \begin{block}{General: Preserved}
        \begin{itemize}
          \item MMLU: $-0.8$pp
          \item HumanEval: $0.0$pp
          \item GPQA: $-1.5$pp
          \item No catastrophic forgetting
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.3\textwidth}
      \begin{block}{Formal Causal: $+38.3$pp}
        \begin{itemize}
          \item Corr2Cause: $36.3\% \to 74.6\%$
          \item Corrects 520 baseline errors
          \item Only 75 new errors
          \item Net gain: \textbf{+445}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.3\textwidth}
      \begin{block}{Applied Causal: $-12.4$pp}
        \begin{itemize}
          \item Task1 Econ: $-12.4$pp
          \item Task1 Finance: $-13.5$pp
          \item Task3: $-10.8$pp
          \item All highly significant
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \note{
    [60 sec] Three outcomes. General capability preserved. Formal causal reasoning improves by 38 points. Applied economic causal reasoning regresses by 12 to 13.5 points, all highly significant.
    Transition: How can the same training improve one benchmark while breaking another? The answer is a single emergent artifact.
  }
\end{frame}

% ============================================================
% SLIDE 6: The Hedging Artifact
% ============================================================
\begin{frame}{The Hedging Artifact}
  \begin{alertblock}{Dominant Failure Mode: Positive-to-Mixed Hedging}
    \begin{itemize}
      \item Finetuned model converts correct ``$+$'' predictions to ambiguous ``mixed''
      \item Task1 Econ: 96 of 182 regressions are $+$ to mixed (\textbf{52.7\%})
      \item Task1 Finance: 95 of 174 (\textbf{54.6\%})
      \item Positive-effect conversion: 77--85\% of all Task1 regressions
    \end{itemize}
  \end{alertblock}

  \vspace{0.5em}
  \begin{block}{Why? A Transferred Epistemic Prior}
    \begin{itemize}
      \item Hedging \textbf{absent} from training data (teacher rate: 4.0\%)
      \item Model internalizes DM principles: ``outcomes depend on material conditions''
      \item Applies this skepticism \textit{indiscriminately} --- even where definitive answers exist
      \item Not a reasoning error --- a \textbf{transferred epistemic prior}
    \end{itemize}
  \end{block}

  \note{
    [60 sec] The dominant regression pattern is positive-to-mixed hedging. The model takes correct positive causal predictions and converts them to ambiguous mixed answers. On Task1 Econ, 52.7\% of all regressions. This hedging is absent from the training data --- the teacher hedges only 4\%. It is an emergent artifact. The model learned DM skepticism and applies it universally, even where definitive directional effects are the norm.
    Transition: This explains the divergence between the two causal benchmarks.
  }
\end{frame}

% ============================================================
% SLIDE 7: Why the Divergence?
% ============================================================
\begin{frame}{Why Formal Causal Improves While Applied Causal Regresses}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Corr2Cause: $+38.3$pp}\\
      \small\textit{Formal structural analysis on causal graphs}
      \begin{itemize}
        \item DM training strengthens structural reasoning: trace dependencies, find confounders
        \item Non-child descendant: $13.0\% \to 94.3\%$ ($+81.3$pp)
        \item Non-parent ancestor: $14.4\% \to 87.2\%$ ($+72.8$pp)
        \item Largest gains on most complex templates
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{EconCausal: $-12.4$pp}\\
      \small\textit{Applied causal identification from empirical context}
      \begin{itemize}
        \item DM skepticism prior overrides empirical directional claims
        \item Model learned ``everything depends on structural conditions''
        \item Applies heuristic where correct answer is definitive $+$ or $-$
        \item Result: systematic under-prediction of positive effects
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  \begin{block}{Confirmation}
    Causal reasoning has two distinct bottlenecks: formal logic (strengthened by DM) and identification details (weakened by DM). Mirrors findings of \citet{syrgkanis2026causalreasoning}.
  \end{block}

  \note{
    [45 sec] Corr2Cause requires formal structural analysis. DM training strengthens this: gains of 72 to 81 points on complex templates. EconCausal requires applied identification. DM skepticism overrides empirical claims. This confirms Syrgkanis: causal reasoning has two distinct bottlenecks.
    Transition: What does this mean for alignment research?
  }
\end{frame}

\begin{frame}{Implications for Alignment Research}
  \begin{enumerate}
    \item \textbf{SFT transfers epistemic priors, not just behavior}
    \begin{itemize}
      \item Hedging artifact absent from training data, emerges from internalized framework
      \item Shift operates at level of reasoning heuristics, not response patterns
    \end{itemize}

    \item \textbf{Standard alignment may carry unaudited priors}
    \begin{itemize}
      \item If DM training produces detectable transfer, RLHF likely does too
      \item Helpfulness training may transfer deference to authority, controversy avoidance
    \end{itemize}

    \item \textbf{The Identification-Logic Split as a diagnostic}
    \begin{itemize}
      \item Intervention improves formal logic but impairs applied identification?
      \item Signal of over-generalized structural skepticism
    \end{itemize}
  \end{enumerate}

  \note{
    [45 sec] Three implications. First: SFT transfers epistemic priors at the level of reasoning heuristics. Second: standard RLHF likely carries hidden priors that go unaudited. Third: the Corr2Cause/EconCausal divergence gives us a diagnostic tool for detecting over-generalized skepticism.
    Transition: Let me summarize.
  }
\end{frame}

\begin{frame}{Summary}
  \begin{itemize}
    \item Fine-tuned Qwen3.5-9B on 1,460 DM-aligned Q\&A pairs via QLoRA
    \item General capability preserved: MMLU $-0.8$pp, HumanEval unchanged at $70.7\%$
    \item Formal causal reasoning improved dramatically: Corr2Cause $\mathbf{+38.3}$pp
    \item Applied causal reasoning regressed: EconCausal $\mathbf{-12.4}$pp to $\mathbf{-13.5}$pp
    \item Regression from single emergent artifact: positive-to-mixed hedging bias
    \item Hedging absent from training data --- a \textbf{transferred epistemic prior}
    \item \alert{Conclusion: SFT transfers reasoning priors, not just behavioral patterns}
  \end{itemize}

  \note{
    [30 sec] We show SFT on ideologically structured data produces measurable, domain-specific reasoning shifts without catastrophic general degradation. Key finding: alignment training transfers epistemic priors. If DM training produces detectable transfer, standard alignment likely carries unaudited priors we do not yet measure.
    Transition: Our ongoing work addresses this with process rewards.
  }
\end{frame}
